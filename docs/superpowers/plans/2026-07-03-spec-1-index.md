> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implementar `GET /posts` con paginación, búsqueda, filtros (status/author), ordenamiento y respuesta paginada, sobre la fundación congelada de Spec 0.

**Architecture:** El endpoint en `app/routers/posts.py` declara y valida los query params, delega la construcción de la query a `list_posts()` en `app/services/post_query.py` (filtros/orden/paginación en un solo lugar, testeable sin HTTP), y envuelve el resultado en el schema `PostList`. Reusa `PostRead`, `Pagination`, `get_db`, `POST_STATUSES` y el handler de errores de Spec 0.

**Tech Stack:** FastAPI, SQLAlchemy 2.0, Pydantic v2, pytest.

## Global Constraints

- Contrato canónico de Post: 11 campos exactos vía `PostRead` (no redefinir).
- Respuesta paginada: `{ "data": [...], "pagination": {total, page, per_page, total_pages} }`.
- Params inválidos → 400 `VALIDATION_ERROR` (formato estándar de Spec 0).
- `per_page`: default 10, rango 1–100. `page`: default 1, ≥ 1.
- `search`: substring case-insensitive en `title` + `content`.
- `status`: por defecto excluye `trash`; `status=<válido>` filtra ese estado exacto.
- `orderby` whitelist: `created_at`, `updated_at`, `title`, `id`. Default `created_at`. `order`: `asc`/`desc`, default `desc`.
- No tocar archivos `[FROZEN]` de la fundación (`database.py`, `models/`, `errors/`, `routers/health.py`).
- Commit por tarea con tests en verde.

```python
from sqlalchemy import func, or_

from app.models.post import Post

ORDERABLE = {
    "created_at": Post.created_at,
    "updated_at": Post.updated_at,
    "title": Post.title,
    "id": Post.id,
}


def list_posts(
    db,
    *,
    search=None,
    status=None,
    author=None,
    orderby="created_at",
    order="desc",
    page=1,
    per_page=10,
):
    query = db.query(Post)

    if status is None:
        query = query.filter(Post.status != "trash")
    else:
        query = query.filter(Post.status == status)

    if search:
        pattern = f"%{search.lower()}%"
        query = query.filter(
            or_(
                func.lower(Post.title).like(pattern),
                func.lower(Post.content).like(pattern),
            )
        )

    if author is not None:
        query = query.filter(Post.author_id == author)

    total = query.count()

    column = ORDERABLE.get(orderby, Post.created_at)
    column = column.desc() if order == "desc" else column.asc()

    items = (
        query.order_by(column)
        .offset((page - 1) * per_page)
        .limit(per_page)
        .all()
    )
    return items, total
```

```python
from math import ceil

from fastapi import APIRouter, Depends, Query
from sqlalchemy.orm import Session

from app.database import get_db
from app.errors.exceptions import ValidationError
from app.models.post import POST_STATUSES
from app.schemas.post import Pagination, PostList
from app.services.post_query import ORDERABLE, list_posts

router = APIRouter(prefix="/posts", tags=["posts"])


@router.get("", response_model=PostList)
def index(
    page: int = Query(1, ge=1),
    per_page: int = Query(10, ge=1, le=100),
    search: str | None = Query(None),
    status: str | None = Query(None),
    author: int | None = Query(None),
    orderby: str = Query("created_at"),
    order: str = Query("desc"),
    db: Session = Depends(get_db),
):
    details = []
    if status is not None and status not in POST_STATUSES:
        details.append({"field": "status", "message": f"Invalid status '{status}'"})
    if orderby not in ORDERABLE:
        details.append({"field": "orderby", "message": f"Invalid orderby '{orderby}'"})
    if order not in ("asc", "desc"):
        details.append({"field": "order", "message": f"Invalid order '{order}'"})
    if details:
        raise ValidationError(details=details)

    items, total = list_posts(
        db,
        search=search,
        status=status,
        author=author,
        orderby=orderby,
        order=order,
        page=page,
        per_page=per_page,
    )
    total_pages = ceil(total / per_page) if total else 0
    return PostList(
        data=items,
        pagination=Pagination(
            total=total, page=page, per_page=per_page, total_pages=total_pages
        ),
    )
```
